# Debug Keystore for Consistent APK Signing
#
# This debug.keystore ensures GitHub Actions builds use the same signing key
# as local development builds, preventing "signatures do not match" errors.
#
# To use this keystore:
# 1. Copy this file to ~/.android/debug.keystore (local dev)
# 2. GitHub Actions will use this file automatically
#
# Keystore Info:
# - Alias: androiddebugkey
# - Password: android
# - Validity: 10,000 days
#
# If you need to regenerate this file, run:
# keytool -genkey -v -keystore debug.keystore -alias androiddebugkey \
#   -keyalg RSA -keysize 2048 -validity 10000 \
#   -storepass android -keypass android \
#   -dname "CN=Android Debug,O=Android,C=US"
#
# Then update the base64 in .github/workflows/build.yml
